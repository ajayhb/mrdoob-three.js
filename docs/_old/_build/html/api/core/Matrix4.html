Example

// simple rig for rotation around 3 axes

var m = new THREE.Matrix4();

var m1 = new THREE.Matrix4();
var m2 = new THREE.Matrix4();
var m3 = new THREE.Matrix4();

var alpha = 0;
var beta = Math.PI;
var gamma = Math.PI/2;

m1.setRotationX( alpha );
m2.setRotationY( beta );
m3.setRotationZ( gamma );

m.multiply( m1, m2 );
m.multiplySelf( m3 );
