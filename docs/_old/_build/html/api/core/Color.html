Example

var colors = [];

for ( i = 0; i < 5000; i ++ ) {

    x = 2000 * Math.random() - 1000;
    colors[ i ] = new THREE.Color( 0xffffff );
    colors[ i ].setHSV( ( x + 1000 ) / 2000, 1.0, 1.0 );

}
